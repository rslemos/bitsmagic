91  			try {
92  				IntegerListOutputStream stream = new IntegerListOutputStream();
93  				Encoder encoder = factory.encoder(stream, outbase);
94  				
95  				for (int i = 0; i < input.length; i++)
96  					encoder.write(input[i], cumulativeCount);
97  				
98  				encoder.flush();
99  				return stream.toIntArray();
100 			} catch (IOException e) {
101 				throw new RuntimeException(e);
102 			}
103 		}

109 				
110 				int[] result = new int[n];

112 				for (int i = 0; i < result.length; i++)
113 					result[i] = decoder.read(cumulativeCount);
114 				
115 				return result;
116 			} catch (IOException e) {
117 				throw new RuntimeException(e);

153 		// for more than 15, n^n would not fit 64-bit longs (let alone the ETA of 12 years)
154 		// needed for the fast encoding function used to test if a permutation was already seen
155 	
156 		private void permutationOf(final int n) {
157 			try {
158 				int base = 1;
159 				
160 				for(int i = 1; i <= n; i++)
161 					base *= i;
162 		
163 				int[] sorted;
164 				
165 				// this will produce 0, 1, 2 ... n-1
166 				{
167 					sorted = even(n);
168 					sorted[0]--;
169 					sorted = toCumulative(sorted);
170 				}
171 		
172 				long prev = 0;
173 				for (int i = 0; i < base; i++) {
174 					IntArrayInputStream stream = new IntArrayInputStream(i);
175 					
176 					Decoder decoder = factory.decoder(stream, base);
177 					int[] count = even(n);
178 							
179 					int[] x = new int[n];
180 					for (int j = 0; j < n; j++) {
181 						int m = decoder.read(toCumulative(count));
182 						count[m]--;
183 						x[j] = m;
184 					}
185 					
186 					// assert that "x" is a new permutation not previously seen
187 					{
188 						// fast, holey encoding of "x" in base n... 
189 						long result = 0;
190 						for (int j = 0; j < n; j++) {
191 							result *= n;
192 							result += x[j];
193 						}
194 						
195 						// ...which nonetheless is guaranteed to be monotonically increasing function of "i"
196 						assertThat(result, is(greaterThan(prev)));
197 						prev = result;
198 					}
199 					
200 					// assert that "x" is a permutation
201 					{
202 						if (i == 0) {
203 							// the very first MUST BE sorted already (if not, the test will fail ON PURPOSE)
204 						} else if (i == base - 1) {
205 							// the very last MUST BE reverse sorted (if not, the test will fail ON PURPOSE)
206 							// since this is the last time using "sorted" we will reverse it mercilessly
207 							reverse(sorted);
208 						} else
209 							sort(x);
210 						
211 						assertThat(x, is(equalTo(sorted)));
212 					}
213 					
214 					// this also asserts that "x" is a permutation
215 					// since no element was returned more than once 
216 					assertThat(count, is(equalTo(new int[n])));
217 				}
218 			} catch (IOException e) {
219 				throw new RuntimeException(e);
220 			}
221 		}
222 		
223 		private static void reverse(int... is) {
224 			for (int i = 0; i < is.length/2; i++) {
225 				int t = is[i];
226 				is[i] = is[is.length - i - 1];
227 				is[is.length - i - 1] = t;
228 			}
229 		}

253 		private void runMultiple0(int externalDigitCount, int externalBase, int internalBase) throws IOException {
254 			int[] externalData = new int[externalDigitCount];
255 			
256 			double fullRange = Math.pow(externalBase, externalData.length);
257 			
258 			if (fullRange <= MAX_SAMPLES)
259 				do op.run(internalBase, externalBase, externalData);
260 				while (inc(externalData, externalBase) == 0);
261 			else {
262 				int amount = (int)((fullRange / MAX_SAMPLES)) | 7;
263 				
264 				if (amount <= Integer.MAX_VALUE / 2) {
265 					do op.run(internalBase, externalBase, externalData);
266 					while (add(externalData, externalBase, amount) == 0);
267 				}
268 			}

376 		public static int inc(int[] digits, int base) {
377 			for (int i = 0; i < digits.length; i++) {
378 				if (++digits[i] < base) return 0;
379 				digits[i] = 0;
380 			}
381 			
382 			return 1;
383 		}
384 		
385 		public static int add(int[] digits, int base, int amount) {
386 			for (int i = 0; i < digits.length; i++) {
387 				digits[i] += amount;
388 				amount = digits[i] / base;
389 				if (amount == 0) return 0;
390 				digits[i] %= base;
391 			}
392 			
393 			return amount;
394 		}

404 		private final Operation op;
405 
406 		Roundtrip(ArithmeticCodecFactory factory, Operation op) {
407 			super(factory);
408 			this.op = op;
409 		}
410 
411 		@Override void run(int internalBase, int externalBase, int[] externalData) throws IOException {
412 			if (externalData[externalData.length - 1] == 0)
413 				return;
414 			
415 			int[] internalData = op.encode(factory, internalBase, externalBase, externalData);
416 			int[] result = op.decode(factory, internalBase, externalData.length, externalBase, internalData);
417 			
418 			assertThat(result, is(equalTo(externalData)));
419 		}

421 		public static abstract class Operation {
422 			public static class NoUnderflow extends Operation {
423 				public static final NoUnderflow NO_UNDERFLOW = new NoUnderflow();
424 				
425 				@Override public int[] encode(ArithmeticCodecFactory factory, int internalBase, int externalBase, int[] externalData) throws IOException {
426 					IntArrayOutputStream sink = new IntArrayOutputStream(internalDigits(internalBase, externalBase, externalData.length));
427 					Encoder encoder = factory.encoder(sink, internalBase);
428 					int[] cumulativeCount = toCumulative(even(externalBase));
429 					for (int d : externalData)
430 						encoder.write(d, cumulativeCount);
431 					encoder.flush();
432 					return sink.toIntArray();
433 				}
434 
435 				@Override public int[] decode(ArithmeticCodecFactory factory, int internalBase, int externalDigits, int externalBase, int[] internalData) throws IOException {
436 					int[] externalData = new int[externalDigits];
437 					IntArrayInputStream source = new IntArrayInputStream(internalData);
438 					Decoder decoder = factory.decoder(source, internalBase);
439 					int[] cumulativeCount = toCumulative(even(externalBase));
440 					for (int i = 0; i < externalData.length; i++) {
441 						externalData[i] = decoder.read(cumulativeCount);
442 					}
443 					return externalData;
444 				}
445 				
446 				public static int internalDigits(int internalBase, int externalBase, int externalDigits) {
447 					return (int) Math.ceil(externalDigits*Math.log(externalBase)/Math.log(internalBase));
448 				}
449 			}

560 				private int internalDigits(int internalBase, int externalBase, int externalDigits) {
561 					// https://en.wikipedia.org/wiki/Entropy_(information_theory)#Definition
562 					
563 					double digits = 0;
564 					
565 					// head prefix
566 					digits += (externalDigits - 1)*(Math.log(externalBase));
567 					// head last symbol
568 					digits += 1*(Math.log(externalBase+1));
569 					// tail prefix
570 					int p_1 = underflow.underflowCumulativeCount[2]/(underflow.underflowCumulativeCount[1] - underflow.underflowCumulativeCount[0]);
571 					digits += underflow.underflowLength*(Math.log(p_1));
572 					// tail last symbol
573 					digits += 1*(Math.log(2));
574 					
575 					digits /= Math.log(internalBase);
576 					
577 					return (int) Math.ceil(digits);
578 				}
