45  public class DelayedZeroIntOutputStreamUnitTest {

60  	}

64  		verify(sink, never()).writeInt(0);
65  	}
66  	
67  	@Test public void delayZeroOnce() throws IOException {
68  		delayNZeros(1);
69  	}
70  	
71  	@Test public void delay32KZeros() throws IOException {
72  		delayNZeros(1 << 15);
73  	}
74  
75  	@Ignore("Meaningful only if DelayedZeroIntOutputStream.zeros is changed to short")
76  	@Test(expected = IllegalStateException.class)
77  	public void delay128KZeros() throws IOException {
78  		delayNZeros(1 << 17);
79  	}
80  
81  	@Ignore("Meaningful only if DelayedZeroIntOutputStream.zeros is changed to short")
82  	@Test public void delay64KZeros() throws IOException {
83  		delayNZeros((1 << 16) - 1);
84  	}
85  	
86  	private void delayNZeros(int n) throws IOException {
87  		for (long i = 0; i < n; i++)
88  			delayedZeroSink.writeInt(0);
89  		delayedZeroSink.writeInt(1);
90  		
91  		verify(sink, times(n)).writeInt(0);
92  		verify(sink).writeInt(1);
93  	}
