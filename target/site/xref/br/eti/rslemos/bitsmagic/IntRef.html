28  package br.eti.rslemos.bitsmagic;
29  
30  /**
31   * Integer to pass by reference. An object of this class allows to pass 
32   * integers to called methods and have changes to their values reflected back 
33   * at the callee. Since objects of this class are not immutable, both methods 
34   * {@code equals} and {@code hashCode} compute their results on object 
35   * references, that is, {@code Object}'s default implementations are not 
36   * overridden.
37   */
38  public class IntRef {
39  	/**
40  	 * Integer value held by this reference.
41  	 */
42  	public int i;
43  
44  	/**
45  	 * Create a new reference with 0 as initial content.
46  	 */
47  	public IntRef() { };
48  
49  	/**
50  	 * Create a new reference with {@code i} as initial content.
51  	 */
52  	public IntRef(int i) { this.i = i; }
53  
54  	/**
55  	 * Create a new reference with {@code i} as initial content. Syntatic sugar 
56  	 * to {@code new IntRef(i)}.
57  	 */
58  	public static IntRef byRef(int i) { return new IntRef(i); }
59  	
60  	@Override public String toString() {
61  		return String.format("%d&", i);
62  	}
63  }
